grad_max <- function(x, dir = 1){
  if(dir == 1){
    map(1:length(x), ~max(x[1:.x])) |>
      simplify() -> out
  }else{
    map(1:length(x), ~max(x[.x:length(x)])) |>
      simplify() -> out
  }
  return(out)
}

rev_diff <- function(x){
  out <- x |> rev() |> diff() |> rev()
  return(out)
}
l_to_r <- apply(tree_mat, 1, grad_max) |> t()

view_line_score <- function(x){
  score <- which((x - x[1])[-1] >= 0)[1]
  if(is.na(score)){
    return(length(x)-1)
  }else{
    return(score)
  }
}
scorer <- function(x, y, mat){
  view_line <- list(
    north = rev(mat[1:(x),y]),
    east = mat[x, (y):ncol(mat)],
    south = mat[x:nrow(mat), y],
    west = rev(mat[x, 1:y])
  )
  
  score <- 
  view_line |>
    map(view_line_score) |>
    reduce(`*`)

  return(score)
}
